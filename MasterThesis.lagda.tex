–í–æ-–≤—Ç–æ—Ä—ã—Ö, –º—ã —Å–≤—è–∂–µ–º $\lambda$-–∞–±—Å—Ç—Ä–∞–∫—Ü–∏–∏ —Å —Ñ—É–Ω–∫—Ü–∏—è–º–∏ –æ—Ç —Å–≤—è–∑–∞–Ω–Ω—ã—Ö –ø–µ—Ä–µ–º–µ–Ω–Ω—ã—Ö,
–∞ $\lambda$-–ø—Ä–∏–º–µ–Ω–µ–Ω–∏—è \dash —Å –≤—ã—á–∏—Å–ª–µ–Ω–∏—è–º–∏ —Ñ—É–Ω–∫—Ü–∏–π –Ω–∞ –∞—Ä–≥—É–º–µ–Ω—Ç–∞—Ö.
–ü—É—Å—Ç—å –≤ —Ç–µ—Ä–º–µ $M$ –µ—Å—Ç—å –ø–æ–¥—Ç–µ—Ä–º $N = (\lambda x : \phi . L)K$. –¢–æ–≥–¥–∞ $M$ 
–Ω–∞—Ö–æ–¥–∏—Ç—Å—è –≤ –æ—Ç–Ω–æ—à–µ–Ω–∏–∏  $\beta$-—Ä–µ–¥—É–∫—Ü–∏–∏ $M \rightarrow_\beta M'$ —Å –Ω–µ–∫–æ—Ç–æ—Ä—ã–º —Ç–µ—Ä–º–æ–º $M'$, 
–ø–æ–ª—É—á–∞–µ–º—ã–º –∏–∑ $M$ –∑–∞–º–µ–Ω–æ–π –∫–∞–∫–æ–≥–æ-–ª–∏–±–æ –æ–¥–Ω–æ–≥–æ –≤—Ö–æ–∂–¥–µ–Ω–∏—è –ø–æ–¥—Ç–µ—Ä–º–∞ $N$ –Ω–∞ –ø–æ–¥—Ç–µ—Ä–º
$L[K/x]$ –ø—Ä–∏ —É—Å–ª–æ–≤–∏–∏ –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ—Å—Ç–∏ –¥–∞–Ω–Ω–æ–π –ø–æ–¥—Å—Ç–∞–Ω–æ–≤–∫–∏: –Ω–∏–∫–∞–∫–∞—è —Å–≤–æ–±–æ–¥–Ω–∞—è –ø–µ—Ä–µ–º–µ–Ω–Ω–∞—è $K$ –Ω–µ
—Å—Ç–∞–Ω–æ–≤–∏—Ç—Å—è –ø–æ—Å–ª–µ —Ç–∞–∫–æ–π –ø–æ–¥—Å—Ç–∞–Ω–æ–≤–∫–∏ —Å–≤—è–∑–∞–Ω–Ω–æ–π. –ü—Ä–∏–≤–µ–¥–µ–º –Ω–µ—Å–∫–æ–ª—å–∫–æ –ø—Ä–∏–º–µ—Ä–æ–≤:
$$(\lambda x : \phi . x) y \rightarrow_\beta y$$
$$(\lambda y : \phi . y) y \rightarrow_\beta y$$
$$(\lambda x : \phi . y)((\lambda y : \psi . y) z) \rightarrow_\beta (\lambda x : \phi . y) z \rightarrow_\beta y$$

–¢–∞–∫–∏–º –æ–±—Ä–∞–∑–æ–º, –ø–æ–ª–æ–∂–∏–≤ –æ—Ç–Ω–æ—à–µ–Ω–∏–µ $\rightarrow$ —Ä–∞–≤–Ω—ã–º –æ–±—ä–µ–¥–∏–Ω–µ–Ω–∏—é $\rightarrow_\alpha$ –∏ $\rightarrow_\beta$,
–ø–æ–ª—É—á–∞–µ–º —Ñ–æ—Ä–º–∞–ª—å–Ω—É—é —Å–∏—Å—Ç–µ–º—É –ø–µ—Ä–µ–ø–∏—Å—ã–≤–∞–Ω–∏—è —Ç–µ—Ä–º–æ–≤ $(\Lambda Terms, \rightarrow_{\alpha \beta}, \twoheadrightarrow_{\alpha \beta})$.

\begin{definition}
    \label{NormalFormDefinition}
    –ü—É—Å—Ç—å $(A, \rightarrow, \twoheadrightarrow)$ \dash —Ñ–æ—Ä–º–∞–ª—å–Ω–∞—è —Å–∏—Å—Ç–µ–º–∞ –ø–µ—Ä–µ–ø–∏—Å—ã–≤–∞–Ω–∏—è —Ç–µ—Ä–º–æ–≤.
    –≠–ª–µ–º–µ–Ω—Ç $a \in A$ —è–≤–ª—è–µ—Ç—Å—è –Ω–æ—Ä–º–∞–ª—å–Ω–æ–π —Ñ–æ—Ä–º–æ–π, –µ—Å–ª–∏ $\forall b \in A (a \not \rightarrow b)$.
\end{definition}

–û–±—â–µ–µ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ \ref{NormalFormDefinition}, –æ–¥–Ω–∞–∫–æ, –Ω–µ –æ—á–µ–Ω—å —Ö–æ—Ä–æ—à–æ —Ä–∞–±–æ—Ç–∞–µ—Ç
–¥–ª—è –Ω–∞—à–µ–π —Å–∏—Å—Ç–µ–º—ã $(\Lambda Terms, \rightarrow_{\alpha \beta}, \twoheadrightarrow_{\alpha \beta})$:
–∫ –ª—é–±–æ–º—É –∑–∞–º–∫–Ω—É—Ç–æ–º—É —Ç–µ—Ä–º—É $M$ –º–æ–∂–Ω–æ –ø—Ä–∏–º–µ–Ω–∏—Ç—å $\alpha$-–∫–æ–Ω–≤–µ—Ä—Å–∏—é. –ü–æ—ç—Ç–æ–º—É –º—ã –±—É–¥–µ–º –≥–æ–≤–æ—Ä–∏—Ç—å 
–æ –Ω–æ—Ä–º–∞–ª—å–Ω–æ–π —Ñ–æ—Ä–º–µ —Å —Ç–æ—á–Ω–æ—Å—Ç—å—é –¥–æ $\alpha$-–∫–æ–Ω–≤–µ—Ä—Å–∏–∏: —Ç–µ—Ä–º $M$ –Ω–∞—Ö–æ–¥–∏—Ç—Å—è –≤ –Ω–æ—Ä–º–∞–ª—å–Ω–æ–π —Ñ–æ—Ä–º–µ
–µ—Å–ª–∏ $\forall N \in \Lambda Term (M \rightarrow_{\alpha \beta} N \Rightarrow M \rightarrow_\alpha N)$.

\begin{definition}
    \label{NormalizationDefinition}
    –≠–ª–µ–º–µ–Ω—Ç $a$ —Å–∏—Å—Ç–µ–º—ã $(A, \rightarrow, \twoheadrightarrow)$ –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è (—Å–ª–∞–±–æ-)–Ω–æ—Ä–º–∞–ª–∏–∑—É–µ–º—ã–º, –µ—Å–ª–∏
    –Ω–∞–π–¥–µ—Ç—Å—è —Ç–∞–∫–æ–π $b \in A$, —á—Ç–æ 
    \begin{enumerate}
        \item $b$ —è–≤–ª—è–µ—Ç—Å—è –Ω–æ—Ä–º–∞–ª—å–Ω–æ–π —Ñ–æ—Ä–º–æ–π.
        \item $a \twoheadrightarrow b$.
    \end{enumerate}
    (–í —á–∞—Å—Ç–Ω–æ—Å—Ç–∏, –≤—Å—è–∫–∞—è –Ω–æ—Ä–º–∞–ª—å–Ω–∞—è —Ñ–æ—Ä–º–∞ —è–≤–ª—è–µ—Ç—Å—è –Ω–æ—Ä–º–∞–ª–∏–∑—É–µ–º–æ–π, –≤ —Å–∏–ª—É —Ä–µ—Ñ–ª–µ–∫—Å–∏–≤–Ω–æ—Å—Ç–∏ –æ—Ç–Ω–æ—à–µ–Ω–∏—è
    $\twoheadrightarrow$.)
\end{definition}

\begin{definition}
    \label{StrongNormalizationDefinition}
    –≠–ª–µ–º–µ–Ω—Ç $a$ —Å–∏—Å—Ç–µ–º—ã $(A, \rightarrow, \twoheadrightarrow)$ –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è —Å–∏–ª—å–Ω–æ-–Ω–æ—Ä–º–∞–ª–∏–∑—É–µ–º—ã–º,
    –µ—Å–ª–∏
    \begin{enumerate}
        \item $a$ —è–≤–ª—è–µ—Ç—Å—è –Ω–æ—Ä–º–∞–ª–∏–∑—É–µ–º—ã–º.
        \item –ù–µ –≤–µ—Ä–Ω–æ —Å–ª–µ–¥—É—é—â–∏–µ —É—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏–µ: —Å—É—â–µ—Å—Ç–≤—É–µ—Ç —Å—á–µ—Ç–Ω–∞—è –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å
        —ç–ª–µ–º–µ–Ω—Ç–æ–≤ $A$ $\{ b_0, b_1, b_2, \dots \}$ —Ç–∞–∫–∞—è, —á—Ç–æ $b_0 = a$ –∏ 
        $b_i \rightarrow b_{i+1}$ –¥–ª—è –ª—é–±–æ–≥–æ –Ω–∞—Ç—É—Ä–∞–ª—å–Ω–æ–≥–æ $i$.
    \end{enumerate}
\end{definition}
–ò–Ω—ã–º–∏ —Å–ª–æ–≤–∞–º–∏, –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ \ref{StrongNormalizationDefinition} –≥–æ–≤–æ—Ä–∏—Ç, —á—Ç–æ $a$ —è–≤–ª—è–µ—Ç—Å—è
—Å–∏–ª—å–Ω–æ-–Ω–æ—Ä–º–∞–ª–∏–∑—É–µ–º—ã–º, –µ—Å–ª–∏ –≤—Å—è–∫–∞—è –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–π $\rightarrow$ —ç–ª–µ–º–µ–Ω—Ç–∞ $a$ –ø—Ä–∏–≤–æ–¥–∏—Ç
–∫ –Ω–æ—Ä–º–∞–ª—å–Ω–æ–π —Ñ–æ—Ä–º–µ –∑–∞ –∫–æ–Ω–µ—á–Ω–æ–µ —á–∏—Å–ª–æ —à–∞–≥–æ–≤.

–ü—Ä–æ–∏–ª–ª—é—Å—Ç—Ä–∏—Ä—É–µ–º –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è \ref{NormalFormDefinition} –∏ \ref{StrongNormalizationDefinition} –ø—Ä–∏–º–µ—Ä–∞–º–∏ –∏–∑
–Ω–∞—à–µ–≥–æ $\lambda$-–∏—Å—á–∏—Å–ª–µ–Ω–∏—è. –¢–µ—Ä–º $(\lambda x : p . y)((\lambda z : q . z) w)$ —è–≤–ª—è–µ—Ç—Å—è —Å–∏–ª—å–Ω–æ –Ω–æ—Ä–º–∞–ª–∏–∑—É–µ–º—ã–º:
–º—ã –ª–∏–±–æ –º–æ–∂–µ–º –ø–µ—Ä–µ–ø–∏—Å–∞—Ç—å –≤–Ω—É—Ç—Ä–µ–Ω–Ω–∏–π –ø–æ–¥—Ç–µ—Ä–º $(\lambda z : q) w$ —Å –ø–æ–º–æ—â—å—é $\beta$-—Ä–µ–¥—É–∫—Ü–∏–∏ –∏ –ø–æ–ª—É—á–∏—Ç—å
—Ç–µ—Ä–º $(x : p . y) w$, –∫–æ—Ç–æ—Ä—ã–π –µ–¥–∏–Ω—Å—Ç–≤–µ–Ω–Ω—ã–º –æ–±—Ä–∞–∑–æ–º $\beta$-—Ä–µ–¥—É—Ü–∏—Ä—É–µ—Ç—Å—è –¥–æ –Ω–æ—Ä–º–∞–ª—å–Ω–æ–π —Ñ–æ—Ä–º—ã $y$, –ª–∏–±–æ
–ª–∏–±–æ —Å—Ä–∞–∑—É –ø–µ—Ä–µ–ø–∏—Å–∞—Ç—å —Å–∞–º–æ–µ –≤–Ω–µ—à–Ω–∏–µ $\lambda$-–ø—Ä–∏–º–µ–Ω–µ–Ω–∏–µ –¥–æ $y$. –î—Ä—É–≥–∏—Ö –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–µ–π $\beta$-—Ä–µ–¥—É–∫—Ü–∏–π
–¥–ª—è —Ç–∞–∫–æ–≥–æ —Ç–µ—Ä–º–∞ –Ω–µ—Ç. –° –¥—Ä—É–≥–æ–π —Å—Ç–æ—Ä–æ–Ω—ã, —Ç–µ—Ä–º $(\lambda x : p . y)((\lambda z : q . zz)(\lambda z : q . zz))$
—è–≤–ª—è–µ—Ç—Å—è —Ç–æ–ª—å–∫–æ —Å–ª–∞–±–æ-–Ω–æ—Ä–º–∞–ª–∏–∑—É–µ–º—ã–º: –∫–∞–∫ –∏ –≤ –ø—Ä–æ—à–ª–æ–º –ø—Ä–∏–º–µ—Ä–µ, –º—ã –º–æ–∂–µ–º —Å—Ä–∞–∑—É —Å—Ä–µ–¥—É—Ü–∏—Ä–æ–≤–∞—Ç—å—Å—è –¥–æ $y$,
–∏–ª–∏ –ø–æ–ø—ã—Ç–∞—Ç—å—Å—è –ø–µ—Ä–µ–ø–∏—Å—ã–≤–∞—Ç—å –ø–æ–¥—Ç–µ—Ä–º $(\lambda z : q . zz)(\lambda z : q . zz)$ –∏ –ø–æ–ª—É—á–∏—Ç—å –±–µ—Å–∫–æ–Ω–µ—á–Ω—É—é –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å
$$(\lambda x : p . y)((\lambda z : q . zz)(\lambda z : q . zz)) \rightarrow_\beta (\lambda x : p . y)((\lambda z : q . zz)(\lambda z : q . zz)) \rightarrow_\beta \dots$$
–°—Ä–∞–∑—É –∂–µ –∑–∞–º–µ—á–∞–µ–º, —á—Ç–æ —Ç–µ—Ä–º $(\lambda z : q . zz)(\lambda z : q . zz)$ –Ω–µ —è–≤–ª—è–µ—Ç—Å—è –Ω–∏ —Å–∏–ª—å–Ω–æ-,
–Ω–∏ —Å–ª–∞–±–æ-–Ω–æ—Ä–º–∞–ª–∏–∑—É–µ–º—ã–º.

\begin{definition}
    \label{ChurchRosserDefinition}
    –°–∏—Å—Ç–µ–º–∞ $(A, \rightarrow, \twoheadrightarrow)$ –æ–±–ª–∞–¥–∞–µ—Ç —Å–≤–æ–π—Å—Ç–≤–æ–º –ß—ë—Ä—á–∞-–†–æ—Å—Å–µ—Ä–∞ (—Å–≤–æ–π—Å—Ç–≤–æ–º –∫–æ–Ω—Ñ–ª—é—ç–Ω—Ç–Ω–æ—Å—Ç–∏),
    –µ—Å–ª–∏ 
    $$\forall a, b, b' \in A \Bigl((a \twoheadrightarrow b \land a \twoheadrightarrow b') \Rightarrow \exists c \in A (b \twoheadrightarrow c \land b' \twoheadrightarrow c) \Bigr)$$
    $$
    \begin{tikzcd}
        & \arrow[ld, two heads] a \arrow[rd, two heads] & \\
        b \arrow[rd, two heads, dashed] & & \arrow[ld, two heads, dashed] b' \\
        & c &
    \end{tikzcd}
    $$
\end{definition}

\begin{lemma}
    \label{UniqueNormalFormLemma}
    –ï—Å–ª–∏ –¥–ª—è —Å–∏—Å—Ç–µ–º—ã $(A, \rightarrow, \twoheadrightarrow)$ –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è —Å–≤–æ–π—Å—Ç–≤–æ –ß—ë—Ä—á–∞-–†–æ—Å—Å–µ—Ä–∞, –∏
    $a$ \dash –Ω–æ—Ä–º–∞–ª–∏–∑—É–µ–º—ã–π —ç–ª–µ–º–µ–Ω—Ç $A$, —Ç–æ –µ–≥–æ –Ω–æ—Ä–º–∞–ª—å–Ω–∞—è —Ñ–æ—Ä–º–∞ $b$ –µ–¥–∏–Ω—Å—Ç–≤–µ–Ω–Ω–∞.
\end{lemma}
\begin{proof}
    –ü—É—Å—Ç—å $a$ —Ä–µ–¥—É—Ü–∏—Ä—É–µ—Ç—Å—è –∫ –¥–≤—É–º –Ω–æ—Ä–º–∞–ª—å–Ω—ã–º —Ñ–æ—Ä–º–∞–º $b$ –∏ $b'$. –¢–æ–≥–¥–∞ –ø–æ —Å–≤–æ–π—Å—Ç–≤—É
    –∫–æ–Ω—Ñ–ª—é—ç–Ω—Ç–Ω–æ—Å—Ç–∏ –Ω–∞–π–¥–µ—Ç—Å—è —Ç–∞–∫–æ–π $c$, —á—Ç–æ $b \twoheadrightarrow c$ –∏ $b' \twoheadrightarrow c$. –ù–æ
    –ª—é–±–∞—è –Ω–æ—Ä–º–∞–ª—å–Ω–∞—è —Ñ–æ—Ä–º–∞ –Ω–∞—Ö–æ–¥–∏—Ç—Å—è –≤ –æ—Ç–Ω–æ—à–µ–Ω–∏–∏ $\twoheadrightarrow$ —Ç–æ–ª—å–∫–æ —Å–∞–º–∞ —Å —Å–æ–±–æ–π.
    –ê –∑–Ω–∞—á–∏—Ç
    $$b = c = b'$$
\end{proof}

\begin{theorem}[\cite{SorUrz06}]
    –°–∏—Å—Ç–µ–º–∞ $(\Lambda Terms, \rightarrow_{\alpha \beta}, \twoheadrightarrow_{\alpha \beta})$ –æ–±–ª–∞–¥–∞–µ—Ç
    —Å–≤–æ–π—Å—Ç–≤–æ–º –ß—ë—Ä—á–∞-–†–æ—Å—Å–µ—Ä–∞.
\end{theorem}

–°—É—â–µ—Å—Ç–≤—É–µ—Ç –Ω–µ—Å–∫–æ–ª—å–∫–æ —Å–ø–æ—Å–æ–±–æ–≤ –∑–∞–∫–æ–¥–∏—Ä–æ–≤–∞—Ç—å –Ω–∞—Ç—É—Ä–∞–ª—å–Ω—ã–µ —á–∏—Å–ª–∞ –≤ $\lambda$-–∏—Å—á–∏—Å–ª–µ–Ω–∏–∏. –û–¥–∏–Ω –∏–∑ —Ç–∞–∫–∏—Ö
—Å–ø–æ—Å–æ–±–æ–≤ \dash –ø–æ—Å—Ç—Ä–æ–∏—Ç—å –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –Ω–∞—Ç—É—Ä–∞–ª—å–Ω–æ–≥–æ $n$ –∫–æ–º–±–∏–Ω–∞—Ç–æ—Ä $[n]$, –Ω–∞–∑—ã–≤–∞–µ–º—ã–π –Ω—É–º–µ—Ä–∞–ª–æ–º –ß—ë—Ä—á–∞,
—Å–ª–µ–¥—É—é—â–∏–º –æ–±—Ä–∞–∑–æ–º:
$$[0] := \lambda f : p \rightarrow p . \lambda x : p . x$$
$$[1] := \lambda f : p \rightarrow p . \lambda x : p . fx$$
$$[2] := \lambda f : p \rightarrow p . \lambda x : p . f (fx)$$
$$[3] := \lambda f : p \rightarrow p . \lambda x : p . f (f (fx))$$
$$ \dots $$
$$[n] := \lambda f : p \rightarrow p. \lambda x : p . \underbrace{f(\dots (f}_{n\ \text{–ø—Ä–∏–º–µ–Ω–µ–Ω–∏–π}\ f} x))$$
$$ \dots $$

\begin{theorem}[\cite{SorUrz06}]
    –î–ª—è –≤—Å—è–∫–æ–π –≤—ã—á–∏—Å–ª–∏–º–æ–π —Ñ—É–Ω–∫—Ü–∏–∏ $f$ –∏–∑ $\mathbb{N}$ –≤ $\mathbb{N}$ —Å—É—â–µ—Å—Ç–≤—É–µ—Ç
    $\lambda$-—Ç–µ—Ä–º $F$ —Ç–∞–∫–æ–π, —á—Ç–æ
    \begin{enumerate}
        \item –ï—Å–ª–∏ $n$ –Ω–µ –ø—Ä–∏–Ω–∞–¥–ª–µ–∂–∏—Ç –æ–±–ª–∞—Å—Ç–∏ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è $f$, —Ç–æ —Ç–µ—Ä–º $F[n]$ –Ω–µ –Ω–æ—Ä–º–∞–ª–∏–∑—É–µ–º.
        \item –ï—Å–ª–∏ $n$ –ø—Ä–∏–Ω–∞–¥–ª–µ–∂–∏—Ç –æ–±–ª–∞—Å—Ç–∏ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è $f$, —Ç–æ $F[n] \twoheadrightarrow_{\alpha \beta} [f(n)]$.
    \end{enumerate}
\end{theorem}

–ü–æ –ª–µ–º–º–µ \ref{UniqueNormalFormLemma} –≤—Å—è–∫–∏–π –Ω—É–º–µ—Ä–∞–ª –ß—ë—Ä—á–∞ –µ–¥–∏–Ω—Å—Ç–≤–µ–Ω–µ–Ω —Å —Ç–æ—á–Ω–æ—Å—Ç—å—é –¥–æ –ø–µ—Ä–µ–∏–º–µ–Ω–æ–≤–∞–Ω–∏—è
—Å–≤—è–∑–∞–Ω–Ω—ã—Ö –ø–µ—Ä–µ–º–µ–Ω–Ω—ã—Ö, –∞ –∑–Ω–∞—á–∏—Ç —Ä–µ–¥—É–∫—Ü–∏—è —Ç–µ—Ä–º–æ–≤ $F[n]$ –∑–∞–¥–∞—ë—Ç –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ—é —Ñ—É–Ω–∫—Ü–∏–æ–Ω–∞–ª—å–Ω–æ–µ –æ—Ç–Ω–æ—à–µ–Ω–∏–µ –Ω–∞
$\mathbb{N}$. –¢–∞–∫–∏–º –æ–±—Ä–∞–∑–æ–º —Å–∏—Å—Ç–µ–º–∞ $(\Lambda Terms, \rightarrow_{\alpha \beta}, \twoheadrightarrow_{\alpha \beta})$
